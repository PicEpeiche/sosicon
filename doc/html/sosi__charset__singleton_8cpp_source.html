44 std::string sosicon::sosi::SosiCharsetSingleton::
45 toIso8859_1( const std::string& str ) {
46  unsigned char const* contable;
47  switch( mCharset ) {
48  case sosi_charset_ansi:
49  case sosi_charset_iso8859_1:
50  case sosi_charset_iso8859_10:
51  return str; // No conversion
52  case sosi_charset_decn7:
53  contable = chartables::DECN7_TO_ISO8859_1;
54  break;
55  case sosi_charset_dosn8:
56  contable = chartables::DOSN8_TO_ISO8859_1;
57  break;
58  case sosi_charset_nd7:
59  contable = chartables::ND7_TO_ISO8859_1;
60  break;
61  case sosi_charset_utf8:
62  return utf8ToIso8859_1( str.c_str() );
63  default:
64  return str; // Unknown charset, no conversion
65  }
66  std::string res = "";
67  std::string::size_type len = str.length();
68  for( std::string::size_type i = 0; i < len; i++ ) {
69  res += static_cast<char>( contable[ static_cast<unsigned char>( str.at( i ) ) ] );
70  }
71  return res;
72 }

74 std::string sosicon::sosi::SosiCharsetSingleton::
75 utf8ToIso8859_1( const char *in ) {
76  std::string out;
77  if ( in == NULL ) {
78  return out;
79  }
80  unsigned int codepoint = 0;
81  while ( *in != 0 )
82  {
83  unsigned char ch = static_cast<unsigned char>( *in );
84  if ( ch <= 0x7f ) {
85  codepoint = ch;
86  }
87  else if ( ch <= 0xbf ) {
88  codepoint = ( codepoint << 6 ) | ( ch & 0x3f );
89  }
90  else if ( ch <= 0xdf ) {
91  codepoint = ch & 0x1f;
92  }
93  else if ( ch <= 0xef ) {
94  codepoint = ch & 0x0f;
95  }
96  else {
97  codepoint = ch & 0x07;
98  }
99  ++in;
100  if ( ( ( *in & 0xc0 ) != 0x80 ) && ( codepoint <= 0x10ffff ) ) {
101  if ( codepoint <= 255 ) {
102  out.append( 1, static_cast<char>( codepoint ) );
103  }
104  }
105  }
106  return out;
107 }
