106 void sosicon::CoordinateCollection::
107 extractPath( sosi::Reference* ref, ISosiElement* referencedElement ) {
108  sosi::SosiElementSearch src;
109  if( referencedElement->getType() == sosi::sosi_element_curve ) {
110  // First segment describes the center point
111  referencedElement->getChild( src, sosi::sosi_element_ne );
112  mCenterPoint = new sosi::SosiNorthEast( src.element() );
113  }
114  while( referencedElement->getChild( src, sosi::sosi_element_ne ) ) {
115  sosi::SosiNorthEast* ne = new sosi::SosiNorthEast( src.element() );
116  sosi::NorthEastList& lst = ref->subtract ? mIslands : mGeom;
117  std::vector<int>& offsets = ref->subtract ? mPartOffsetsIslands : mPartOffsetsGeom;
118  int& pointCount = ref->subtract ? mNumPointsIslands : mNumPointsGeom;
119  if( ref->reverse ) {
120  ne->reverse();
121  }
122  lst.push_back( ne );
123  offsets.push_back( pointCount );
124  pointCount += ne->getNumPoints();
125  ne->expandBoundingBox( mXmin, mYmin, mXmax, mYmax );
126  }
127 }
