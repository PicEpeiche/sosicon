59 bool sosicon::
60 isClockwise( std::vector<ICoordinate*>::iterator& begin, std::vector<ICoordinate*>::iterator& end ) {
61  double edgeSum = 0.0;
62  for( std::vector<ICoordinate*>::iterator i = begin; i != end; i++ ) {
63  ICoordinate* v1 = *i;
64  ICoordinate* v2 = *( i + 1 != end ? i + 1 : begin );
65  edgeSum += ( v2->getE() - v1->getE() ) * ( v2->getN() + v1->getN() );
66  }
67  return edgeSum > 0;
68 }
