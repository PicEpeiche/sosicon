93 void sosicon::shape::Shapefile::
94 buildShpElement( ISosiElement* sosi ) {
95 
96  CoordinateCollection cc;
97  cc.discoverCoords( sosi );
98  ICoordinate* c = 0;
99 
100  Int32Field recordNumber;
101  recordNumber.i = ++mRecordNumber;
102 
103  Int32Field shapeType;
104  shapeType.i = 3;
105 
106  Int32Field numParts;
107  numParts.i = cc.getNumPartsGeom();
108 
109  Int32Field numPoints;
110  numPoints.i = cc.getNumPointsGeom();
111 
112  int byteLength = 44 + ( 4 * numParts.i ) + ( 16 * numPoints.i );
113 
114  Int32Field contentLength;
115  contentLength.i = byteLength / 2;
116 
117  double xMin = cc.getXmin();
118  double yMin = cc.getYmin();
119  double xMax = cc.getXmax();
120  double yMax = cc.getYmax();
121 
122  mXmin = std::min( mXmin, xMin );
123  mYmin = std::min( mYmin, yMin );
124  mXmax = std::max( mXmax, xMax );
125  mYmax = std::max( mYmax, yMax );
126 
127  int o = 0;
128 
129  ShxIndex shxIndex;
130  shxIndex.offset.i = 50 + ( mShpBufferSize / 2 );
131  shxIndex.length.i = contentLength.i;
132  mShxOffsets.push_back( shxIndex );
133 
134  if( 0 == mShpBufferSize ) {
135  mShpBufferSize = byteLength + 8;
136  mShpBuffer = new char [ mShpBufferSize ];
137  }
138  else {
139  char* oldBuffer = mShpBuffer;
140  o = mShpBufferSize;
141  mShpBufferSize += ( byteLength + 8 );
142  mShpBuffer = new char [ mShpBufferSize ];
143  std::copy( oldBuffer, oldBuffer + o, mShpBuffer );
144  delete [ ] oldBuffer;
145  }
146 
147  // Record header
148  byteOrder::toBigEndian( recordNumber.b, &mShpBuffer[ o + 0 ], 4 ); // Record serial
149  byteOrder::toBigEndian( contentLength.b, &mShpBuffer[ o + 4 ], 4 ); // Record content length
150 
151  // Record content
152  byteOrder::toLittleEndian( shapeType.b, &mShpBuffer[ o + 8 ], 4 ); // Shape type
153  byteOrder::doubleToLittleEndian( xMin, &mShpBuffer[ o + 12 ] ); // Box minX
154  byteOrder::doubleToLittleEndian( yMin, &mShpBuffer[ o + 20 ] ); // Box minY
155  byteOrder::doubleToLittleEndian( xMax, &mShpBuffer[ o + 28 ] ); // Box maxX
156  byteOrder::doubleToLittleEndian( yMax, &mShpBuffer[ o + 36 ] ); // Box maxY
157  byteOrder::toLittleEndian( numParts.b, &mShpBuffer[ o + 44 ], 4 ); // NumParts
158  byteOrder::toLittleEndian( numPoints.b, &mShpBuffer[ o + 48 ], 4 ); // NumPoints
159 
160  o += 52;
161 
162  int part = -1;
163 
164  while( cc.getNextOffsetInGeom( part ) ) {
165  Int32Field offset;
166  offset.i = part;
167  byteOrder::toLittleEndian( offset.b, &mShpBuffer[ o ], sizeof( offset.b ) ); // Shape type
168  o += 4;
169  }
170 
171  while( cc.getNextInGeom( c ) ) {
172  byteOrder::doubleToLittleEndian( c->getE(), &mShpBuffer[ o ] );
173  byteOrder::doubleToLittleEndian( c->getN(), &mShpBuffer[ o + 8 ] );
174  o += 16;
175  }

180  DbfRecord rec;
181  std::string field;
182  std::string data;
183  ISosiElement* child = 0;
184  sosi::SosiElementSearch src;
185  while( sosi->getChild( src ) ) {
186  child = src.element();
187  data = stringUtils::trim( child->getData() );
188  int length = data.size();
189  if( !data.empty() && length < 254 ) {
190  field = child->getName();
191  if( mDbfFieldLengths.find( field ) != mDbfFieldLengths.end() ) {
192  mDbfFieldLengths[ field ] = std::max( mDbfFieldLengths[ field ], length );

196  }

204 buildDbf() {
205 
206  Int16Field recordLength;
207  recordLength.i = 0;
208 
209  for( DbfFieldLengths::iterator i = mDbfFieldLengths.begin(); i != mDbfFieldLengths.end(); i++ ) {
210  recordLength.i += i->second;
211  }
212 
213  Int16Field headerLength;
214  headerLength.i =
215 
216  /* Fixed header size */ sizeof( mDbfHeader ) +
217  /* Field description array */ ( mDbfFieldLengths.size() * 32 ) +
218  /* Terminator */ 1;
219 
220  mDbfBufferSize =
221 
222  /* Field description array */ ( mDbfFieldLengths.size() * 32 ) +
223  /* Terminator */ 1 +
224  /* Record structure */ ( recordLength.i * mDbfRecordSet.size() ) +
225  /* EOF */ 1 ;
226 
227  mDbfBuffer = new char [ mDbfBufferSize ];
228 
229  time_t rawTime;
230  struct tm* timeInfo;
231  time( &rawTime );
232  timeInfo = localtime( &rawTime );
233  Int32Field numRecords;
234  numRecords.i = mDbfRecordSet.size();
235 
236  mDbfHeader[ 0 ] = 0x03; // Version number
237  mDbfHeader[ 1 ] = char( timeInfo->tm_year ); // Year of last update
238  mDbfHeader[ 2 ] = char( timeInfo->tm_mon + 1 ); // Month of last update
239  mDbfHeader[ 3 ] = char( timeInfo->tm_mday ); // Day of last update
240 
241  byteOrder::toLittleEndian( numRecords.b, &mDbfHeader[ 4 ], 4 ); // Number of records
242  byteOrder::toLittleEndian( headerLength.b, &mDbfHeader[ 8 ], 2 ); // Length of header structure
243  byteOrder::toLittleEndian( recordLength.b, &mDbfHeader[ 10 ], 2 ); // Length of record
244 
245  // Reserved or N/A
246  for( int i = 12; i < 32; i++ ) {
247  mDbfHeader[ i ] = 0x00;
248  }
249 
250  int o = 0;
251 
252  for( DbfFieldLengths::iterator i = mDbfFieldLengths.begin(); i != mDbfFieldLengths.end(); i++ ) {
253 
254  std::string fieldName = i->first;
255  fieldName.resize( 10, ' ' );
256  const char* sz = fieldName.c_str();
257  std::copy( sz, sz + 11, &mDbfBuffer[ o ] );
258 
259  // Field data type (char)
260  mDbfBuffer[ o + 11 ] = 'C';
261 
262  // Field data address (N/A)
263  for( int i = 12; i < 16; i++ ) {
264  mDbfBuffer[ o + i ] = 0x00;
265  }
266  mDbfBuffer[ o + 16 ] = char( i->second );
267 
268  // Reserved or N/A
269  for( int i = 17; i < 32; i++ ) {
270  mDbfBuffer[ o + i ] = 0x00;
271  }
272  o += 32;
273  }
274 
275  // Terminator
276  mDbfBuffer[ o++ ] = 0x0d;
277  mDbfBuffer[ o++ ] = 0x20;
278 
279  // Records
280  for( DbfRecordSet::iterator i = mDbfRecordSet.begin(); i != mDbfRecordSet.end(); i++ ) {
281  int recNumber = 0;
282  int fldOffset = 0;
283  char* recordBuffer = new char[ recordLength.i ];
284  DbfRecord& rec = *i;
285  for( DbfFieldLengths::iterator j = mDbfFieldLengths.begin(); j != mDbfFieldLengths.end(); j++ ) {
286  std::string fieldName = j->first;
287  std::string fieldValue;
288  if( rec.find( fieldName ) != rec.end() ) {
289  fieldValue = rec[ fieldName ];
290  }
291  int fieldLength = mDbfFieldLengths[ fieldName ];
292  fieldValue.resize( fieldLength, ' ' );
293  const char* sz = fieldValue.c_str();
294  std::copy( sz, sz + fieldLength, &recordBuffer[ fldOffset ] );
295  fldOffset += fieldLength;
296  }
297  std::copy( recordBuffer, recordBuffer + recordLength.i, &mDbfBuffer[ o ] );
298  delete [ ] recordBuffer;
299  o += recordLength.i;
300  }
301 
302  // End of file
303  mDbfBuffer[ o ] = 0x1a;
304 }
