349 void sosicon::shape::Shapefile::
350 buildDbfHeader( Int16Field recordLength ) {
351 
352  Int16Field headerLength;
353  headerLength.i =
354  /* Fixed header size */ sizeof( mDbfHeader ) +
355  /* Field description array */ ( mDbfFieldLengths.size() * 32 ) +
356  /* Terminator */ 1;
357 
358  time_t rawTime;
359  struct tm* timeInfo;
360  time( &rawTime );
361  timeInfo = localtime( &rawTime );
362  Int32Field numRecords;
363  numRecords.i = mDbfRecordSet.size();
364 
365  mDbfHeader[ 0 ] = 0x03; // Version number
366  mDbfHeader[ 1 ] = char( timeInfo->tm_year ); // Year of last update
367  mDbfHeader[ 2 ] = char( timeInfo->tm_mon + 1 ); // Month of last update
368  mDbfHeader[ 3 ] = char( timeInfo->tm_mday ); // Day of last update
369 
370  byteOrder::toLittleEndian( numRecords.b, &mDbfHeader[ 4 ], 4 ); // Number of records
371  byteOrder::toLittleEndian( headerLength.b, &mDbfHeader[ 8 ], 2 ); // Length of header structure
372  byteOrder::toLittleEndian( recordLength.b, &mDbfHeader[ 10 ], 2 ); // Length of record
373

376  mDbfHeader[ i ] = 0x00;
