125 std::string sosicon::ConverterSosi2shp::
126 makeBasePath( std::string objTypeName ) {
127  std::string candidatePath, dir, tit, ext;
128  if( !mCmd->mOutputFile.empty() ) {
129  candidatePath = mCmd->mOutputFile;
130  }
131  else if( !mCmd->mDestinationDirectory.empty() ) {
132  utils::getPathInfo( mCurrentSourcefile, dir, tit, ext );
133  candidatePath = utils::stripTrailingSlash( mCmd->mDestinationDirectory ) + "/" + tit + ext;
134  }
135  else {
136  candidatePath = mCurrentSourcefile;
137  }
138  dir.clear();
139  utils::getPathInfo( candidatePath, dir, tit, ext );
140  std::string subdir = dir + tit;
141  char separator;
142 
143  if( mCmd->mMakeSubDir ) {
144  if( !utils::fileExists( subdir ) ) {
145 #ifdef _WIN32
146  mkdir( subdir.c_str() );
147 #else
148  mkdir( subdir.c_str(), S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH );
149 #endif
150  }
151  separator = '/';
152  candidatePath = subdir + separator + objTypeName;
153  }
154  else {
155  separator = '_';
156  candidatePath = dir + tit + separator + objTypeName;
157  }
158  int sequence = 0;
159 
160  while( utils::fileExists( candidatePath + ".shp" ) ||
161  utils::fileExists( candidatePath + ".shx" ) ||
162  utils::fileExists( candidatePath + ".dbf" ) ||
163  utils::fileExists( candidatePath + ".prj" ) )
164  {
165  std::stringstream ss;
166  ss << subdir << separator << objTypeName << "_" << std::setw( 2 ) << std::setfill( '0' ) << ++sequence;
167  candidatePath = ss.str();
168  }
169 
170  return candidatePath;
171 }

173 void sosicon::ConverterSosi2shp::
174 run( bool* cancel ) {
175  bool userAborted = false;
176  for( std::vector<std::string>::iterator f = mCmd->mSourceFiles.begin(); f != mCmd->mSourceFiles.end() && !( cancel && *cancel ); f++ ) {
177  mCurrentSourcefile = *f;
178  if( !utils::fileExists( mCurrentSourcefile ) ) {
179  sosicon::logstream << mCurrentSourcefile << " not found!\n";
180  }
181  else {
182  sosicon::logstream << "Reading " << mCurrentSourcefile << "\n";
183  Parser p;
184  char ln[ 1024 ];
185  std::ifstream ifs( ( mCurrentSourcefile ).c_str() );
186  int n = 0;
187  while( !ifs.eof() ) {
188  if( ++n % 100 == 0 ) {
189  if( cancel && *cancel ) {
190  userAborted = true;
191  break;
192  }
193  sosicon::logstream << "\rParsing line " << n;
194  }
195  memset( ln, 0x00, sizeof ln );
196  ifs.getline( ln, sizeof ln );
197  p.ragelParseSosiLine( ln );
198  }
199  p.complete();
200  ifs.close();
201  if( !userAborted ) {
202  sosicon::logstream << "\r" << n << " lines parsed \n";
203  sosicon::logstream << "Building shape file...\n";
204  ISosiElement* root = p.getRootElement();
205  makeShp( root, cancel );
206  }
207  }
208  }
