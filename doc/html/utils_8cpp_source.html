22 string sosicon::utils::
23 className2FileName( const std::string &className )
24 {
25  string clsL = toLower( className );
26  string res = "";
27  string::size_type len = className.length();
28  for( string::size_type i = 0; i < len; i++ )
29  {
30  if( i > 0 && // 1) Not first character?
31  className.at( i ) != clsL.at( i ) && // 2) Uppercase letter?
32  ( i + 1 ) < len && // 3) Not last character in class name?
33  ( className.at( i + 1 ) == clsL.at( i + 1 ) || // 4) Following character is lower case
34  className.at( i - 1 ) == clsL.at( i - 1 ) ) ) // _OR_ previous character is lower case?
35  {
36  res += "_"; // Then: Separate words with underscore
37  }
38  res += clsL.at( i );
39  }
40  return res;
41 }
