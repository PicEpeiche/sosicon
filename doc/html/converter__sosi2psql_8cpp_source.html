145 std::string sosicon::ConverterSosi2psql::
146 buildInsertStatement( Wkt wktGeom,
147  std::string dbSchema,
148  std::string dbTable ) {
149 
150  std::string sqlComposite;
151  std::string geometryType = utils::wktToStr( wktGeom );
152 
153  if( !geometryType.empty() && mRowsListCollection[ wktGeom ]->size() > 0 ) {
154 
155  std::string sqlInsert;
156  std::string sqlValues;
157 
158  RowsList::iterator itrRows;
159  FieldsList::iterator itrFields;
160 
161  FieldsList* f = mFieldsListCollection[ wktGeom ];
162  RowsList* r = mRowsListCollection[ wktGeom ];
163 
164  std::string geomField = dbTable + "_geom";
165  std::string geomName = utils::toLower( geometryType );
166 
167  for( itrFields = f->begin(); itrFields != f->end(); itrFields++ ) {
168  if( sqlInsert.empty() ) {
169 
170  sqlInsert = "INSERT INTO "
171  + dbSchema
172  + "."
173  + dbTable
174  + "_"
175  + geomName
176  + " (" + itrFields->first;
177  }
178  else {
179  sqlInsert += ( "," + itrFields->first );
180  }
181  }
182  sqlInsert += ") VALUES\n";
183  int rowCount = 0;
184  RowsList::size_type len = r->size();
185  sosicon::logstream << " > Processing 0 of " << len << sosicon::flush;
186  for( itrRows = r->begin(); itrRows != r->end(); itrRows++ ) {
187  std::map<std::string,std::string>* row = *itrRows;
188  if( !sqlValues.empty() && ++rowCount % 50000 == 0 ) {
189  sqlValues = sqlValues.substr( 0, sqlValues.length() - 2 );
190  sqlValues += ";\n";
191  sqlComposite += ( sqlInsert + sqlValues );
192  sqlValues.clear();
193  }
194  if( rowCount % 1000 == 0 ) {
195  sosicon::logstream << "\r > Processing " << rowCount << " of " << len << sosicon::flush;
196  }
197  sqlValues += "(";
198  for( itrFields = f->begin(); itrFields != f->end(); itrFields++ ) {
199  std::string key = itrFields->first;
200  if( row->find( key ) == row->end() ) {
201  ( *row )[ key ] = "";
202  }
203  std::string val = utils::trim( ( *row )[ key ] );
204  if( val.empty() ) {
205  sqlValues += itrFields->second.isNumeric() ? "NULL," : "'',";
206  }
207  else if( key == geomField ) {
208  sqlValues += val + ",";
209  }
210  else if( itrFields->second.isNumeric() ) {
211  sqlValues += utils::sqlNormalize( val ) + ",";
212  }
213  else {
214  sqlValues += "'" + utils::sqlNormalize( val ) + "',";
215  }
216  }
217  sqlValues.erase( sqlValues.size() - 1 );
218  sqlValues += "),\n";
219  }
220  sosicon::logstream << "\r > " << rowCount << " " << geomName << "s processed \n" << sosicon::flush;
221  sqlValues.erase( sqlValues.size() - 2 );
222  sqlValues += ";\n";
223  sqlComposite += ( sqlInsert + sqlValues );
224  }
225  return sqlComposite;
226 }

286 getSrid( ISosiElement* sosiTree ) {
287 
288  // Path: .HODE/..TRANSPAR/...KOORDSYS
289  ISosiElement* head = 0, * transpar = 0, * coordsys = 0;
290 
291  sosi::SosiElementSearch srcHead( sosi::sosi_element_head );
292  sosi::SosiElementSearch srcTranspar( sosi::sosi_element_transpar );
293  sosi::SosiElementSearch srcCoordsys( sosi::sosi_element_coordsys );
294 
295  if( sosiTree->getChild( srcHead ) ) {
296  head = srcHead.element();
297  }
298  if( head && head->getChild( srcTranspar ) ) {
299  transpar = srcTranspar.element();
300  }
301  if( transpar && transpar->getChild( srcCoordsys ) ) {
302  coordsys = srcCoordsys.element();
303  }
304 
305  std::string srid;
306 
307  if( coordsys ) {
308  std::stringstream ss;
309  ss << coordsys->getData();
310  int sysCode;
311  ss >> sysCode;
312  sosi::SosiTranslationTable tt;
313  sosi::CoordSys cs = tt.sysCodeToCoordSys( sysCode );
314  srid = cs.srid();
315  sosicon::logstream << "Coordinate system: " << cs.displayString() << "\n";
316  }
317  else {
318  sosicon::logstream << "No KOORDSYS code found in sosi file.\nDefaults to 23 (EPSG:25833 - ETRS89 / UTM zone 33N)\n";
319  srid = "25833";
320  }
321 
322  return srid;

559 void sosicon::ConverterSosi2psql::
560 run( bool* ) {
561 
562  mFieldsListCollection[ wkt_point ] = new FieldsList();
563  mFieldsListCollection[ wkt_linestring ] = new FieldsList();
564  mFieldsListCollection[ wkt_polygon ] = new FieldsList();
565 
566  mRowsListCollection[ wkt_point ] = new RowsList();
567  mRowsListCollection[ wkt_linestring ] = new RowsList();
568  mRowsListCollection[ wkt_polygon ] = new RowsList();
569 
570  std::string sridDest = mCmd->mSrid.empty() ? "4326" : mCmd->mSrid;
571 
572  std::string dbSchema = mCmd->mDbSchema.empty() ? "sosicon" : mCmd->mDbSchema;
573  std::string dbTable = mCmd->mDbTable.empty() ? "object" : mCmd->mDbTable;
574  std::string geomField = dbTable + "_geom";
575 
576  ( *mFieldsListCollection[ wkt_point ] )[ geomField ] = Field();
577  ( *mFieldsListCollection[ wkt_linestring ] )[ geomField ] = Field();
578  ( *mFieldsListCollection[ wkt_polygon ] )[ geomField ] = Field();
579 
580  for( std::vector<std::string>::iterator f = mCmd->mSourceFiles.begin(); f != mCmd->mSourceFiles.end(); f++ ) {
581  mCurrentSourcefile = *f;
582  if( !utils::fileExists( mCurrentSourcefile ) ) {
583  sosicon::logstream << mCurrentSourcefile << " not found\n";
584  }
585  else {
586  sosicon::logstream << "Reading " << mCurrentSourcefile << "\n";
587  Parser p;
588  char ln[ 1024 ];
589  std::ifstream ifs( mCurrentSourcefile.c_str() );
590  int n = 0;
591  while( !ifs.eof() ) {
592  if( ++n % 100 == 0 ) {
593  sosicon::logstream << "\rParsing line " << n;
594  }
595  memset( ln, 0x00, sizeof ln );
596  ifs.getline( ln, sizeof ln );
597  p.ragelParseSosiLine( ln );
598  }
599  p.complete();
600 
601  ifs.close();
602  sosicon::logstream << "\r" << n << " lines parsed \n";
603  sosicon::logstream << "Building postGIS export...\n";
604  ISosiElement* root = p.getRootElement();
605  makePsql( root, sridDest, dbSchema, dbTable );
606  }
607  }
608  writePsql( sridDest, dbSchema, dbTable );
609  cleanup();
610  sosicon::logstream << "Done!\n";
611 }
