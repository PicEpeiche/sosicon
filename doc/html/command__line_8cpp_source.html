47 void sosicon::CommandLine::
48 parse( int argc, char* argv[] ) {
49 
50  bool inputExpected = true;
51  std::string inputLine = "";
52  if( !mIsTtyIn ) {
53  while( std::cin ) {
54  std::cin >> inputLine;
55  mSourceFiles.push_back( inputLine );
56  }
57  }
58 
59  for( int i = 1; i < argc; i++ ) {
60  std::string param( argv[ i ] );
61  if( param.at( 0 ) != '-' ) {
62  while( i < argc ) {
63  mSourceFiles.push_back( argv[ i ] );
64  i++;
65  }
66  }
67  else {
68 
69  if( "-a" == param ) {
70  mAppend = true;
71  }
72  else if( "-id" == param && argc > ( ++i ) ) {
73  mFilterSosiId = utils::explode( ',', argv[ i ] );
74  }
75  else if( "-d" == param && argc > ( ++i ) ) {
76  mDestinationDirectory = argv[ i ];
77  }
78  else if( "-h" == param ) {
79  mIncludeHeader = true;
80  }
81  else if( "-o" == param && argc > ( ++i ) ) {
82  mOutputFile = argv[ i ];
83  }
84  else if( "-t" == param && argc > ( ++i ) ) {
85  mObjTypes = utils::explode( ',', argv[ i ] );
86  }
87  else if( "-g" == param && argc > ( ++i ) ) {
88  mGeomTypes = utils::explode( ',', argv[ i ] );
89  }
90  else if( "-f" == param && argc > ( ++i ) ) {
91  mFieldSelection = utils::explode( ',', argv[ i ] );
92  }
93  else if( "-v" == param ) {
94  mVerbose = 1;
95  }
96  else if( "-V" == param ) {
97  mVerbose = 2;
98  }
99  else if( "-2shp" == param ) {
100  mCommand = param;
101  }
102  else if( "-2tsv" == param ) {
103  mCommand = param;
104  }
105  else if( "-stat" == param ) {
106  mCommand = param;
107  }
108  else if( "-help" == param ) {
109  inputExpected = false;
110  mCommand = param;
111  outputHelpText();
112  break;
113  }
114  else if( "-w" == param ) {
115  inputExpected = false;
116  mCommand = param;
117  outputDisclaimer();
118  break;
119  }
120  else if( "-c" == param ) {
121  inputExpected = false;
122  mCommand = param;
123  outputLicense();
124  break;
125  }
126  }
127  }
128 
129  if( mCommand.empty() || ( inputExpected && mSourceFiles.size() == 0 ) ) {
130  std::cout << "Missing input parameters.\n";
131  std::cout << "Type sosicon -help for instructions.\n\n";
132  return;
133  }
134 }
