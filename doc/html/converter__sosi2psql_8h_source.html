54  class Field {
55  std::string::size_type mMaxLength;
56  std::string::size_type mMinLength;
57  bool mIsNumeric;
58  public:
59  bool isNumeric() {
60  if( mMaxLength == mMinLength ) {
61  // Treat fixed-length numerical data as character field, since they
62  // are in fact non-arithmetic types types like dates, phone numbers
63  // or serial numbers.
64  mIsNumeric = false;
65  }
66  return mIsNumeric;
67  }
68  std::string::size_type length() {
69  return mMaxLength;
70  }
71  Field() {
72  mIsNumeric = true;
73  mMaxLength = 0;
74  }
75  Field( std::string& str ) {
76  mIsNumeric = true;
77  mMaxLength = 0;
78  mMinLength = std::numeric_limits<std::string::size_type>::max();
79  expand( str );
80  }
81  std::string::size_type expand( std::string& str ) {
82  std::string::size_type len = str.length();
83  mMinLength = std::min( mMinLength, len );
84  mMaxLength = std::max( mMaxLength, len );
85  if( mIsNumeric ) {
86  mIsNumeric = utils::isNumeric( str );
87  }
88  return mMaxLength;
89  }
90  };
