55 void sosicon::byteOrder::
56 doubleToLittleEndian( double from, char* to ) {
57  // Using Dieter Lückings solution posted on StackOverflow Dec 8 '13:
58  // http://stackoverflow.com/a/20453208
59  uint8_t sign = 0;
60  uint16_t exponent = 0;
61  uint64_t fraction = 0;
62  const uint16_t exp_base = 0x03ff;
63  double dbl = from;
64  if( std::signbit( dbl ) ) {
65  sign = 1;
66  dbl = -dbl;
67  }
68  if( std::isinf( dbl ) ) {
69  exponent = 0x7ff;
70  }
71  else if( std::isnan( dbl ) ) {
72  exponent = 0x7ff;
73  fraction = 0x8000000000000;
74  }
75  else {
76  int e;
77  double f = frexp( dbl, &e );
78  // Adjust exponent to account for the implicit, leading 1
79  exponent = int16_t( e + exp_base - 1 );
80  unsigned bits = 0;
81  while( f ) {
82  f *= 2;
83  fraction <<= 1;
84  if( 1 <= f ) {
85  fraction |= 1;
86  f -= 1;
87  }
88  bits++;
89  }
90  fraction = ( fraction << ( 53 - bits ) ) & ( ( uint64_t( 1 ) << 52 ) - 1 );
91  }
92  uint8_t data[ sizeof( double ) ];
93  for( unsigned i = 0; i < 6; ++i ) {
94  data[ i ] = fraction & 0xff;
95  fraction >>= 8;
96  }
97  data[ 6 ] = ( exponent << 4 ) | fraction;
98  data[ 7 ] = ( sign << 7 ) | ( exponent >> 4 );
99  double result = * ( double* ) &data;
100  if( result != from && ( std::isnan( from ) != std::isnan( result ) ) ) {
101  std::reverse( data, data + sizeof( double ) );
102  }
103  std::copy( data, data + sizeof( double ), to );
104 }
